        function atualizarHorario() {
            const agora = new Date();
            const horas = agora.getHours();
            const minutos = agora.getMinutes();
            const proximaAula = new Date(agora.getTime() + (2 * 60 * 60 * 1000)); // +2 horas
            const horasProxima = proximaAula.getHours();
            const minutosProxima = proximaAula.getMinutes();
            document.getElementById('proximaAula').innerText = `Próxima aula: ${horasProxima.toString().padStart(2, '0')}:${minutosProxima.toString().padStart(2, '0')}`;
        }

        document.addEventListener("DOMContentLoaded", function() {
            atualizarHorario();
            setInterval(atualizarHorario, 60 * 1000); // Atualiza a cada minuto
        });
